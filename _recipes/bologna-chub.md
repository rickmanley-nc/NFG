  - group: Main
    items:
      - item: bologna chub
        amount: 3
        unit: pounds
        note: whole chub, scored in a crosshatch pattern
      - item: yellow mustard
        amount: 2
        unit: tablespoons
        note: binder for seasoning
  - group: Seasoning Rub
    items:
      - item: brown sugar
        amount: 2
        unit: tablespoons
      - item: paprika
        amount: 1
        unit: tablespoon
      - item: garlic powder
        amount: 1
        unit: teaspoon
      - item: onion powder
        amount: 1
        unit: teaspoon
      - item: kosher salt
        amount: 1
        unit: teaspoon
      - item: black pepper
        amount: 1
        unit: teaspoon
  - group: Optional Glaze
    items:
      - item: barbecue sauce
        amount: 1/2
        unit: cup
      - item: honey
        amount: 1
        unit: tablespoon
instructions:
  - group: Preparation
    steps:
      - step: "[PLACEHOLDER] Remove packaging from the bologna chub and score the outside in a shallow crosshatch pattern"
      - step: "[PLACEHOLDER] Lightly coat the chub with yellow mustard to help the rub adhere"
      - step: "[PLACEHOLDER] Mix seasoning ingredients and apply evenly over all sides"
  - group: Smoking
    steps:
      - step: "[PLACEHOLDER] Preheat smoker to 250°F using oak, hickory, or cherry wood"
      - step: "[PLACEHOLDER] Place the seasoned bologna chub directly on the grate"
      - step: "[PLACEHOLDER] Smoke for 2-3 hours until the exterior is deep mahogany and edges begin to crisp"
      - step: "[PLACEHOLDER] For a glazed finish, brush with barbecue sauce and honey during the final 20 minutes"
  - group: Serving
    steps:
      - step: "[PLACEHOLDER] Remove from smoker and rest for 10 minutes before slicing"
      - step: "[PLACEHOLDER] Slice into rounds or cubes and serve warm with crackers, pickles, or sandwiches"
