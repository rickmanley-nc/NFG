ingredients:
  - group: Dumpling Filling
    items:
      - item: ground pork
        amount: 1
        unit: lb
      - item: napa cabbage
        amount: 2
        unit: cups
        note: finely chopped and squeezed dry
      - item: fresh ginger
        amount: 1
        unit: tablespoon
        note: grated
      - item: garlic
        amount: 4
        unit: cloves
        note: minced
      - item: soy sauce
        amount: 2
        unit: tablespoons
      - item: sesame oil
        amount: 1
        unit: tablespoon
      - item: rice wine vinegar
        amount: 1
        unit: tablespoon
      - item: white pepper
        amount: 1/2
        unit: teaspoon
      - item: green onions
        amount: 4
        unit: stalks
        note: thinly sliced
  - group: Lasagna Assembly
    items:
      - item: wonton wrappers
        amount: 60-70
        unit: wrappers
        note: square or round, thawed if frozen
      - item: ricotta cheese
        amount: 15
        unit: oz
      - item: mozzarella cheese
        amount: 3
        unit: cups
        note: shredded, divided
      - item: egg
        amount: 1
      - item: Italian seasoning
        amount: 1
        unit: teaspoon
  - group: Sauce
    items:
      - item: marinara sauce
        amount: 24
        unit: oz
        note: about 3 cups
      - item: chili garlic sauce
        amount: 2
        unit: tablespoons
        note: Huy Fong or similar
      - item: chicken broth
        amount: 1/2
        unit: cup
  - group: Garnish
    items:
      - item: green onions
        note: sliced, for topping
      - item: sesame seeds
        note: for topping
      - item: chili oil
        note: optional, for drizzling
instructions:
  - group: Filling Preparation
    steps:
      - step: "[PLACEHOLDER] Finely chop the napa cabbage, sprinkle with salt, let sit 10 minutes, then squeeze out excess moisture in a clean towel"
      - step: "[PLACEHOLDER] In a large bowl, combine ground pork, cabbage, ginger, garlic, soy sauce, sesame oil, rice wine vinegar, white pepper, and green onions"
      - step: "[PLACEHOLDER] Mix until well combined — do not overmix or the filling will be tough"
  - group: Sauce and Cheese Prep
    steps:
      - step: "[PLACEHOLDER] Mix marinara sauce with chili garlic sauce and chicken broth — this is your fusion sauce"
      - step: "[PLACEHOLDER] In a separate bowl, combine ricotta, egg, 1 cup mozzarella, and Italian seasoning"
  - group: Assembly
    steps:
      - step: "[PLACEHOLDER] Preheat oven to 375°F"
      - step: "[PLACEHOLDER] Spread 1/2 cup of sauce on the bottom of a 9x13 baking dish"
      - step: "[PLACEHOLDER] Layer wonton wrappers to cover the bottom, overlapping slightly"
      - step: "[PLACEHOLDER] Spread half the dumpling filling evenly over the wrappers"
      - step: "[PLACEHOLDER] Add another layer of wonton wrappers"
      - step: "[PLACEHOLDER] Spread all of the ricotta mixture over the wrappers"
      - step: "[PLACEHOLDER] Add another layer of wonton wrappers"
      - step: "[PLACEHOLDER] Spread remaining dumpling filling"
      - step: "[PLACEHOLDER] Top with final layer of wonton wrappers"
      - step: "[PLACEHOLDER] Pour remaining sauce over the top, making sure edges are covered (wonton wrappers need moisture to cook)"
      - step: "[PLACEHOLDER] Top with remaining 2 cups mozzarella"
  - group: Baking
    steps:
      - step: "[PLACEHOLDER] Cover tightly with foil and bake for 25 minutes"
      - step: "[PLACEHOLDER] Remove foil and bake an additional 15-20 minutes until cheese is golden and bubbly"
      - step: "[PLACEHOLDER] Let rest 10-15 minutes before cutting — this is important for clean slices"
      - step: "[PLACEHOLDER] Garnish with sliced green onions, sesame seeds, and a drizzle of chili oil"
notes: "[PLACEHOLDER] Ricky will add personal notes here. This fusion recipe combines the flavors of Chinese dumplings with the comfort of Italian lasagna — wonton wrappers replace traditional noodles for a lighter texture. The key is squeezing the moisture out of the cabbage and making sure the sauce covers the wonton wrapper edges so they cook through."
